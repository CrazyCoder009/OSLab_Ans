#include <iostream>
using namespace std;
int main()
{
    int a; 
    cin>>a;
    int b[a];
    for(int i=0;i<a;i++) 
      	cin>>b[i];
    int c;
    cin>>c;
    int j=0;
    for( j=0;j<a;j++) 
    {
     	if(b[j]==c)
      	{
        	cout<<c<<" is present at location "<<j+1; 
        	break;
      	}
    }
    if(j==a)
    {
    	cout<<c<<" is not present in array";
    }
    return 0;
}
