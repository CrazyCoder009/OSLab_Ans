#include <iostream>
using namespace std;
int main() 
{
	int size,count=0;;
  	cin>>size;
  	int arr[size];
  	for(int i=0;i<size;i++)
      	cin>>arr[i];
 	int find,flag=0;
  	cin>>find;
  	for(int i=0;i<size;i++)
    {
    	if(find==arr[i])
        {
         	cout<<find<<" is present at location "<<i+1<<endl;
          	flag=1;
        	count++;
        }
    }
  	if(flag==1)
  		cout<<find<<" is present "<<count<<" times in array";
  	else
      	cout<<find<<" is not present in array";
  	return 0;
}
